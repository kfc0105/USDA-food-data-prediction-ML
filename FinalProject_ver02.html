def extract_ingr(jsonfile):
    """Return the json file to python variable"""
    with open(jsonfile) as f:
        d = json.load(f)
    return [x['ingredients'] for x in d]

def extract_cat(jsonfile):
    """Return the json file to python variable"""
    with open(jsonfile) as f:
        d = json.load(f)
    return [x['brandedFoodCategory'] for x in d]

def extract_desc(jsonfile):
    """Return the json file to python variable"""
    with open(jsonfile) as f:
        d = json.load(f)
    return [x['description'] for x in d]

def predict(data):
    #docs_new has to be one dimentional list
    docs_new = data
    X_new_counts = cv.transform(docs_new)
    predicted = clf.predict(X_new_counts)
    result = []
    for i in predicted:
        result.append(target[1][i])
    return result
